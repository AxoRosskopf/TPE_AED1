\section{Constantes}

% AUXILIARES USADOS 

\aux{cantidadDeMinasJugadas}{t:tablero, j:jugadas}{\ent}{\sum_{i=0}^{|j|-1} {\IfThenElse{(t[x(j[i])[y(j[i]) = true)}{1}{0}}}

\aux{cantidadDeMinasTablero}{ t : \tablero}{\ent}{\sum_{i = 0}^{|t|-1}\sum_{j = 0}^{|t|- 1} 
    $if (t[i][j] = \True ) then 1 else 0 $ fi}
    
\aux{x}{jugada: pos \times \ent}{\ent}{(jugada_0)_0}{}

\aux{y}{jugada: pos \times \ent}{\ent}{(jugada_0)_1}{}


% PREDICADOS USADOS

\pred{esAdyacente}{X: \pos,  P:\pos}{(\longitud{X_0-P_0}=1 \wedge (\longitud{X_1-P_1}=1  \vee X_1=P_1 ) \bigvee (X_0=P_0 \wedge \longitud{X_1-P_1}=1 )}

\pred{posicionEsValida}{t: tablero, p: pos}{0\leq p_0 < \longitud{t} \y 0\leq p_1 < \longitud{t}}{}

\pred{repiteJugada}{j:jugadas}{	(\forall i : \ent)( (0 \leq i < |j|) \implicaLuego (\exists k: \ent) ((0 \leq k \leq |j|)
    \y (i\neq k)) \yLuego (x(j[i]) = x(j[k]) \y y(j[i]) = y(j[k]))}{}

\pred{predPerdio}{t : \tablero, j : \jugadas}{
   		cantidadDeMinas( t , j) = 1
    }

\pred{predGano}{t : \tablero, j : \jugadas}{
   		|t|*|t|-cantidadDeMinasTablero(t)=|j|
    }
    
\pred{tableroValido}{t : \tablero}{
		(\forall i : \ent)(( (0 \leq i < |t|) \y (|t| = |t[0]|))\implicaLuego |t[i]| = |t[0]|)
}

\pred{posEnJugadas}{p: \pos, j:\jugadas}{(\exists i: \ent) ((0\leq i < \longitud{j}) \yLuego (x(j[i])=p_0 \wedge y(j[i])=p_1)) }{}

\pred{posEstaEnJugadas}{t:\tablero, p: \pos, j:\jugadas}{(\exists i: \ent) ((0\leq i < \longitud{j}) \yLuego (x(j[i])=p_0 \wedge y(j[i])=p_1) \wedge minasAdyacentes (t,p)=J[i]_1) }{}


\pred{posEnBanderitas}{p: \pos, b:\banderitas}{(\exists i: \ent) ((0\leq i < \longitud{b}) \yLuego ((b[i])_0=p_0 \wedge (b[i])_1=p_1))}{}

\pred{caminoLibrePlus}{t : \tablero,p0 : \pos ,p1 : \pos}
{ tableroValido(t) \y posicionValida(p_{0}) \y posicionValida(p_{1})\y \\
((\exists s : \TLista{\pos}) \\ 
((\forall i : \ent)(0 \leq i < |s|-1\implicaLuego (minasAdyacentes(t, s[i]) = 0 \y esAdyacente(s[i] ,s[i+1])) \\
\yLuego p0 = s[0] \y p1 = s[|s| - 1])}

\subsection*{Ejercicio 1}

\aux{minasAdyacentes}{t:\tablero, p: \pos}{\ent}{\sum_{i=0}^{\longitud{t}-1}\sum_{j=0}^{\longitud{t}-1} \IfThenElse{esAdyacente((i, j), p) \wedge t[i][j]=true)}{1}{0}} 


\subsection*{Ejercicio 2}
\pred{juegoValido}{t: tablero, j: jugadas}{(\forall i : \ent)( (0 \leq i < |j|) $\implicaLuego$ (posicionEsValida(j[i]_0) \bigwedge ($j[i]_1$ = minasAdyacentes (t, j[i]_0)) \bigwedge  $\neg$ repiteJugada(j) \bigwedge \\($0\leq cantidadDeMinas(t, j) \leq 1)$)} {}

\subsection*{Ejercicio 3}

\begin{proc}{plantarBanderita}{in t:\tablero, in j:\jugadas, in p: \pos, inout b: \banderitas}{}
    \pre{tableroValido(t) \wedge posicionEsValida (p) \wedge \neg(posEnJugadas(p, j) \wedge \neg (posEnBanderitas (p,b)) \wedge b=b_0}
    \post{pos \in b \wedge (\forall i:\ent)(0\leq i < \longitud{b_0}) \implicaLuego b_0[i] \in b}
\end{proc}

\subsection*{Ejercicio 4}

\begin{proc}{perdio}{\In i: \tablero, \In j: \jugadas, \Out res: \bool}{}
	\pre{  juegoValido(t , j) \y tableroValido(t) }
    \post{ res : \True \leftrightarrow predPerdio(t , j)}
\end{proc}



\subsection*{Ejercicio 5}
\begin{proc}{gano}{\In i: \tablero, \In j: \jugadas, \Out res: \bool}{}
	\pre{  juegoValido(t , j) \y tableroValido(t) \y \neg (predPerdio(t , j)) }
    \post{ res : \True \leftrightarrow predGano(t , j)}
\end{proc}


\subsection*{Ejercicio 6}
\begin{proc}{jugar}{\In t : \tablero , \In b : \banderitas, \In p : \pos , \Inout j : \jugadas}{}
    \pre { tableroValido(t) \y juegoValido(t , j) \y \neg predPerdio(t , j) \y \neg predGano(t , j) \y \neg posEnJugadas(p , j) \\ \y \neg posEnBanderitas(p , b) \y j = j_{0}}
    \post{(\forall i : \ent)((0 \leq i < |j_{0}|) \implicaLuego(j_{o}[i] \in j \y ((\exists k : \ent)(0 \leq k < |j| \yLuego (x(j[k]) = p_{0} \y y(j[k]) = p_{1} \y j[k][1] = minasAdyacentes(t , j))))}
\end{proc}
